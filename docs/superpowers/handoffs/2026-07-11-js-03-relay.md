```ts
const SEGMENT_SPLIT = /[.!?;|]+/; // commas deliberately do NOT split
function normalizePhrase(text: string): string {
  const tokens = text.toLowerCase().match(/[\p{L}\p{N}]+/gu);
  return tokens === null ? "" : tokens.join(" ");
}
export function extractMaterialSegments(markdown: string): { raw: string; phrase: string }[] {
  // per line: strip /^\s*(?:[#>]+\s*)+/ then /^\s*(?:[-*+]|\d{1,3}[.)])\s+/ then /[*_`]/g,
  // split on SEGMENT_SPLIT, trim ŌåÆ raw, normalizePhrase ŌåÆ phrase, skip phrase===""
}
export function verifyMarkdownCoverage(input): MarkdownCoverageVerdict {
  const proposed = extractMaterialSegments(input.markdown);
  if (proposed.length === 0) return { ok: false, unverifiedSpans: [] }; // DEFEAT 2
  // corpus: extractMaterialSegments over each source.content + each confirmedText,
  //   each corpus phrase held as ` ${phrase} ` (space-padded, per-segment ŌĆö no
  //   cross-line false adjacency)
  // per unique proposed phrase: needle = ` ${phrase} `; unverified unless SOME corpus
  //   string .includes(needle)  ŌåÉ word-boundary contiguous-run matching
  // unverifiedSpans: echo RAW segment text (cap ~200 chars), slice(0, UNVERIFIED_SPANS_MAX);
  //   truncation must never flip the verdict
}
```

Semantics: a proposed segment passes iff its full normalized token run appears
contiguously inside ONE corpus segment. Sub-phrase of one corpus sentence ŌåÆ pass;
recombination across segments ŌåÆ fail. Unicode tokenization fixes ├ēcole; whole-segment
matching removes the caps/digit + first-word heuristics entirely.
