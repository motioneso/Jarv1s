```typescript
import type { FastifyReply } from "fastify";

import { handleRouteError } from "@jarv1s/module-sdk";

/**
 * Shared settings-route error handler. Wraps the module-sdk handler with a mapper that
 * surfaces account-state errors (pending approval / deactivated) as 403s. Extracted from
 * locale-routes / persona-routes / source-behavior-routes, which were byte-identical (#299).
 */
export function handleSettingsRouteError(error: unknown, reply: FastifyReply) {
  return handleRouteError(error, reply, {
    mappers: [
      (e, r) => {
        if (e instanceof Error) {
          const code = (e as Error & { code?: string }).code;
          if (code === "account_pending_approval" || code === "account_deactivated") {
            return r.code(403).send({ error: e.message, code });
          }
        }
        return undefined;
      }
    ]
  });
}
```

### Task 3: Add backup/restore database-password validation

**Files:**

- Modify: `scripts/restore-database.ts:40-47,89` (same)
- Test: `tests/integration/release-hardening.test.ts` (mirror the existing missing-username test ~line 531)

**Interfaces:**

- Consumes: existing `createBackupPlan` / `createRestorePlan` exports — behavior change only (new throw).

      after the existing "rejects ... missing a username" test (~line 547):

```typescript
it("rejects backup and restore connection strings missing a password", () => {
  // An empty password silently forwards an empty PGPASSWORD, which can fall through to
  // trust/peer auth on a misconfigured host; require the URL to carry a password (#299).
  expect(() =>
    createBackupPlan({
      connectionString: "postgres://jarv1s@db.example.test:5432/jarv1s",
      outputFile: "backups/jarv1s-test.dump"
    })
  ).toThrow("Backup database URL must include a password");

  expect(() =>
    createRestorePlan({
      backupFile: "backups/jarv1s-test.dump",
      connectionString: "postgres://jarv1s@db.example.test:5432/jarv1s"
    })
  ).toThrow("Restore database URL must include a password");
});
```
