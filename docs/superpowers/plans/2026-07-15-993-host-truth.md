# #993 Host Truth (Slice 1) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Give Advanced host setup one-click Herdr install, a derived system-health summary, and remove the fake "Log level" setting — Slice 1 of #993 only (profile/account/email is Slice 2, out of scope).

**Architecture:** A new fixed, argument-free, single-flight install executor sits behind a new admin-only `POST /api/admin/host/install` route (mirrors the existing `GET /api/admin/host/diagnostics` 3-phase DB/outside-DB/DB pattern). Frontend reuses the multiplexer status query's already-existing `herdrInstalled` field — no new polling needed for install state.

**Tech Stack:** Fastify + `@jarv1s/db` DataContextDb, Kysely, `@jarv1s/shared` DTO/schema contracts, React + TanStack Query.

**Prerequisite (already done):** Root-workspace detection unification — `packages/ai/src/adapters/root-workspace.ts` `isRootWorkspaceConfigured(env)`, wired into `decideMultiplexer` and `makeMultiplexerUsableProbe`. Committed `ea9c2d77`. No task here re-does this.

## Global Constraints

- Admin-only: every new route runs `assertAdminUser` before any other work (host-diagnostics-routes.ts pattern) — a non-admin must not be able to distinguish "not available" from "not admin".
- The installer accepts **no** request-derived command, URL, path, version, or argument, ever. Fixed `execFile("bash", [INSTALL_SCRIPT_PATH])` only.
- Installer stdout/stderr is never read into the API process, returned in a response, or logged — only a structured `{ ok, timedOut }` result crosses the process boundary.
- Install is single-flight (module-level lock) and bounded by a timeout.
- `DataContextDb`-only repo access; DB work and the exec call never share a transaction (exec runs strictly between two `withDataContext` calls).
- `AccessContext` stays `{ actorUserId, requestId }` — no new fields.
- Audit write goes through `SettingsRepository.insertAuditEvent` directly (metadata-only: `{ timedOut }`, no stdout/stderr, no paths).
- Any new response field must be added to its Fastify schema (`additionalProperties:false`) or `fast-json-stringify` silently drops it — cover with an `app.inject` test.
- No new module claims these routes (module-isolation) — they go in `PLATFORM_UNGUARDED_ROUTES` instead, admin-gated in-handler.
- No migration needed — `app.admin_audit_events` already exists and `insertAuditEvent` is already public.
- Spec: `docs/superpowers/specs/2026-07-15-settings-host-account-truth.md` (Approved), Locked Decisions 1–3.

```ts
// apps/api/src/herdr-install-port.ts
import { execFile } from "node:child_process";
import { promisify } from "node:util";
import { join } from "node:path";
import { fileURLToPath } from "node:url";

import type { FastifyInstance } from "fastify";

const execFileAsync = promisify(execFile);

const REPO_ROOT = fileURLToPath(new URL("../../..", import.meta.url));
const INSTALL_SCRIPT_PATH = join(REPO_ROOT, "scripts", "install-herdr.sh");
const INSTALL_TIMEOUT_MS = 60_000;

export interface HerdrInstallDependencies {
  readonly install: () => Promise<{ ok: boolean; timedOut: boolean }>;
}

/**
 * #993 — fixed, non-shell, argument-free Herdr install executor. Runs ONLY
 * scripts/install-herdr.sh via execFile (argv array, never a shell string) —
 * no request can ever supply a command, path, or argument. Single-flight:
 * concurrent callers await the same in-flight run. The lock is process-local;
 * a database advisory lock would be needed if API replicas are introduced.
 */
export function createHerdrInstallPort(
  server: Pick<FastifyInstance, "log">
): HerdrInstallDependencies {
  let installInFlight: Promise<{ ok: boolean; timedOut: boolean }> | null = null;

  async function runInstall(): Promise<{ ok: boolean; timedOut: boolean }> {
    let timer: NodeJS.Timeout | undefined;
    try {
      await Promise.race([
        execFileAsync("bash", [INSTALL_SCRIPT_PATH]),
        new Promise((_resolve, reject) => {
          timer = setTimeout(
            () => reject(new Error("herdr install timed out")),
            INSTALL_TIMEOUT_MS
          );
        })
      ]);
      return { ok: true, timedOut: false };
    } catch (error) {
      const timedOut = (error as Error).message === "herdr install timed out";
      // Never log error.message/stdout/stderr here — may echo script output.
      server.log.error({ timedOut }, "herdr install failed (#993)");
      return { ok: false, timedOut };
    } finally {
      if (timer) clearTimeout(timer);
    }
  }

  return {
    install: async () => {
      if (!installInFlight) {
        installInFlight = runInstall().finally(() => {
          installInFlight = null;
        });
      }
      return installInFlight;
    }
  };
}
```

```ts
import type { FastifyInstance, FastifyReply, FastifyRequest } from "fastify";

import type { AccessContext, DataContextDb, DataContextRunner, User } from "@jarv1s/db";
import { HttpError } from "@jarv1s/module-sdk";
import { postHerdrInstallRouteSchema, type HerdrInstallResultDto } from "@jarv1s/shared";

import type { SettingsRepository } from "./repository.js";
import type { GetChatMultiplexerStatus } from "./routes.js";

export interface HerdrInstallDependencies {
  readonly install: () => Promise<{ ok: boolean; timedOut: boolean }>;
}

export interface HostInstallRoutesDependencies {
  readonly dataContext: DataContextRunner;
  readonly resolveAccessContext: (request: FastifyRequest) => Promise<AccessContext>;
  readonly repository: SettingsRepository;
  readonly getChatMultiplexerStatus?: GetChatMultiplexerStatus;
  readonly herdrInstall?: HerdrInstallDependencies;
  readonly assertAdminUser: (scopedDb: DataContextDb, userId: string) => Promise<User>;
  readonly requireRequestId: (request: FastifyRequest) => string;
  readonly handleRouteError: (error: unknown, reply: FastifyReply) => unknown;
}

/**
 * POST /api/admin/host/install — admin-only, single-flight, fixed-script-only
 * Herdr install (#993). Phase 1 (DB context) authorizes and reads the current
 * multiplexer setting. Phase 2 (OUTSIDE any DB context) runs the bounded,
 * argument-free installer. Phase 3 (a second DB context) writes the audit
 * event and re-derives fresh status. Installer stdout/stderr never crosses
 * into this handler — only a structured ok/timedOut result does.
 */
export function registerHerdrInstallRoutes(
  server: FastifyInstance,
  dependencies: HostInstallRoutesDependencies
): void {
  server.post(
    "/api/admin/host/install",
    { schema: postHerdrInstallRouteSchema },
    async (request, reply) => {
      try {
        const accessContext = await dependencies.resolveAccessContext(request);
        const multiplexer = await dependencies.dataContext.withDataContext(
          accessContext,
          async (scopedDb) => {
            await dependencies.assertAdminUser(scopedDb, accessContext.actorUserId);
            if (!dependencies.herdrInstall) {
              throw new HttpError(503, "Herdr install is not available");
            }
            const { multiplexer: mux } =
              await dependencies.repository.getChatMultiplexerSetting(scopedDb);
            return mux;
          }
        );

        const installer = dependencies.herdrInstall as HerdrInstallDependencies;
        const result = await installer.install();

        return await dependencies.dataContext.withDataContext(accessContext, async (scopedDb) => {
          await dependencies.repository.insertAuditEvent(scopedDb, {
            actorUserId: accessContext.actorUserId,
            action: result.ok ? "host.herdr_install.succeeded" : "host.herdr_install.failed",
            targetType: "host",
            targetId: null,
            metadata: { timedOut: result.timedOut },
            requestId: dependencies.requireRequestId(request)
          });

          const status = (await dependencies.getChatMultiplexerStatus?.(multiplexer)) ?? {
            available: { tmux: false, herdr: false },
            herdrInstalled: false,
            active: null,
            activeSource: null,
            envOverride: null
          };

          const state: HerdrInstallResultDto["state"] = result.ok
            ? "installed"
            : result.timedOut
              ? "timeout"
              : "failed";

          return { state, herdrInstalled: status.herdrInstalled } satisfies HerdrInstallResultDto;
        });
      } catch (error) {
        return dependencies.handleRouteError(error, reply);
      }
    }
  );
}
```
